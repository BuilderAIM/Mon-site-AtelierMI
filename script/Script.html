// Fade-in au scroll pour éléments .fade-in const faders = document.querySelectorAll('.fade-in'); const appearOptions = { threshold: 0.2, rootMargin: "0px 0px -50px 0px" }; const appearOnScroll = new IntersectionObserver((entries, observer) => { entries.forEach(entry => { if (!entry.isIntersecting) return; entry.target.classList.add('appear'); observer.unobserve(entry.target); }); }, appearOptions); faders.forEach(fader => { appearOnScroll.observe(fader); }); // Optionnel : nav plus opaque au scroll window.addEventListener('scroll', () => { const nav = document.querySelector('.navbar'); if (window.scrollY > 50) { nav.style.backgroundColor = 'rgba(255,255,255,1)'; } else { nav.style.backgroundColor = 'rgba(255,255,255,0.8)'; } });
